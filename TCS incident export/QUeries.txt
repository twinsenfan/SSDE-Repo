                ,POLICY_CATEGORIES
                ,PA_RP_SERVICES.DESCRIPTION
                ,[DESTINATIONS]

				,PA_REPO_USERS.LONG_NAME

          AND PA_EVENTS_201402.SERVICE_ID = PA_RP_SERVICES.ID AND
          PA_REPO_USERS.ID = PA_MNG_USERS.GUID or PA_MNG_USERS.GUID = NULL
